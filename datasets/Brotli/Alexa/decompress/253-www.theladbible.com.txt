    /**
     * Loads images, scripts, styles, iframes, videos and audios asynchronously.
     * @param {NodeList} [nodes] - A NodeList of elements. By default, it is the result of `querySelectorAll('[data-aload]')`.
     * @param cb - callback to be called once loaded
     * @returns {NodeList}
     *
     * @link https://github.com/pazguille/aload
     */
    function aload(cb) {
      'use strict';
    
      var nodes = window.document.querySelectorAll('[data-aload]');
    
      if (nodes.length === undefined) {
        nodes = [nodes];
      }
    
      var i = 0, len = nodes.length, node;
    
      for (i; i < len; i += 1) {
        node = nodes[i];
        node[ node.tagName !== 'LINK' ? 'src' : 'href' ] = node.getAttribute('data-aload');
        node.removeAttribute('data-aload');
        cb(node);
      }
    
      return nodes;
    }
    
    function imageFallback(node, fallbackSize) {
      var segments = node.src.split("/");
      var newSrc = "";
      segments.forEach(function (segment, index) {
        if(segment.indexOf("content") !== -1
                && segments[(index + 1)].indexOf("x") !== -1) {
          if((fallbackSize !== null)) {
            segments[(index + 1)] = fallbackSize;
          }else{
            segments.splice((index + 1), 1);
          }
        }
      });
    
      segments.forEach(function (segment, index) {
        newSrc += segment;
    
        if(index !== (segments.length -1)) {
          newSrc += "/";
        }
      });
    
      if(newSrc !== ""){
        // Only want to remove the picture element and replace it's sources
        if(node.parentNode.tagName === "PICTURE") {
          var img = document.createElement('img');
          img.src = newSrc;
          node.parentNode.parentNode.appendChild(img);
          node.parentNode.parentNode.removeChild(node.parentNode);
          img.classList.add('image--loaded');
        }else{ // Otherwise we're potentially removing parent elements
          if(page.pageType === 'content') {
            node.parentNode.classList.add('image--loaded');
            if(node.parentNode.hasChildNodes()) {
              var childNodes = node.parentNode.childNodes;
              for (var i = 0; i < childNodes.length; i += 1) {
                if(childNodes[i].tagName === "SOURCE") {
                  childNodes[i].remove();
                }
              }
            }
    
            node.classList.add('image--loaded');
            node.src = newSrc;
          }
        }
      }
    }
