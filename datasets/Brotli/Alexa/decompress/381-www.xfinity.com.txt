jQuery(function ($) {
    $('[data-trigger="link-tracking"]').on('click', function (e) {
        TrackLink(this, $(e.target).data('destination') , e, null, true);

(function() {
/* This is an attempt to fix the fact that the video player tries to center the video 
     * but generally has the wrong layout information when doing so (since the page is)
     * only responsive due to the javascript it is built on.
     *
     * To use it, add the vidLink class to the link you want the listener attached to. If 
     * there is more than one link present on the page, append a number in order after the
     * class name. Example: vidLink1, vidLink2, vidLink3, and so on.
     */

    /* Create an object to hold the link references */
    var videoLinks = {};
    /* When a link is clicked, we repeatedly check for an element with the visible class,
     * and scroll to it if/when it appears.
     */
    var videoClickHandler = function videoClickHandler() {
        var videoTimeout = 0;
        var vidCheckHandler = function vidCheckHandler() {
            var activeVideo = document.querySelector('.x-video--visible');
            if (activeVideo) {
                activeVideo.scrollIntoView();
                var $el = $(activeVideo),
                    elOffset = $el.offset().top,
                    elHeight = $el.height(),
                    windowHeight = $(window).height(),
                    offset;
                if (elHeight < windowHeight) {
                     offset = elOffset - ((windowHeight / 2) - (elHeight / 2));
                }
                 else {
                     offset = elOffset;
                }
                $('html, body').animate({scrollTop:offset}, 0);
                return;
            }
            videoTimeout++;
            if (videoTimeout < 200) {
                setTimeout(vidCheckHandler, 100);
            }
        };
        vidCheckHandler();
    };

    var videoLinkInit = function videoLinkInit(linkCount) {
        /* Links are numbered 1-X, we dynamically generate the references based on link count */
        if (linkCount > 1 && linkCount !== undefined) {
            for (var i=linkCount; i>0; i--) {
                videoLinks['link' + i] = document.querySelector('.vidLink' + i);
                videoLinks['link' + i].addEventListener('click', videoClickHandler, true);
                videoLinks['link' + i].addEventListener('touchend', videoClickHandler, true);
            }
        } else {
            videoLinks.link = document.querySelector('.vidLink');
            videoLinks.link.addEventListener('click', videoClickHandler, true);
            videoLinks.link.addEventListener('touchend', videoClickHandler, true);
        }
    };

    $(window).on('load', function(){
        videoLinkInit();
    });
})();
</script>

    threshhold || (threshhold = 250);

      var context = scope || this;
      var now = +new Date,
        args = arguments;
