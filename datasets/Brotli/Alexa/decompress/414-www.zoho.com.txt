function sbg(){
  var buttons = document.getElementsByClassName('sbg-button');
  var getElement = document.querySelector('[data-sbg-network="facebook"]');               // select the first fb button using custome attr in the document
 // var getUrl=getElement.getAttribute('data-sbg-url');                   // get the [data-sbg-url ]

 // var getHeight=getElement.getAttribute('data-sbg-height');                    
  for(var i=0;i<buttons.length;i++){
    var button = buttons[i]
    var socialUrl = buttons[i];                  
    var attUrl = document.createAttribute("data-sbg-url");                    // Create a "class" attribute
    var attWidth = document.createAttribute("data-sbg-width");                  // Create a "class" attribute
    var attHeight = document.createAttribute("data-sbg-height");                  // Create a "class" attribute
    attUrl.value = getUrl;   
    attWidth.value = getWidth;   
    attHeight.value = getHeight;                                  
    socialUrl.setAttributeNode(attUrl);                           // Set the value of the class attribute
    socialUrl.setAttributeNode(attWidth);  
    socialUrl.setAttributeNode(attHeight);                            

    if(button.hasAttribute('data-sbg-isBinded')){continue}
    button.setAttribute('data-sbg-isBinded','true')
  }

 

  function buildUrl(url, parameters){
    var qs = "";
    for(var key in parameters) {
      var value = parameters[key];
      if(!value){continue}
      value = value.toString().split('\"').join('"');
      qs += key + "=" + encodeURIComponent(value) + "&";
    
    }
    if (qs.length > 0){
      qs = qs.substring(0, qs.length-1); //chop off last "&"
      url = url + "?" + qs;
    }
    return url;
  }

}
