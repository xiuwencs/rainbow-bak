        // SHOW / HIDE MODULES
        for (j=0; j < targets.length; j++) {
          var module = targets[j][0];

          if (targetId != module.id) {
            $(module).addClass('hidden');
          } else {
            $(module).removeClass('hidden');
          }
        }

        // ADD CURRENT CLASS TO TAB
        for (k=0; k < tabs.length; k++) {
          if (tabs[k] == this) {
            $(tabs[k]).addClass('current');
          } else {
            $(tabs[k]).removeClass('current');
          }
        }
      });
    }

    function lazyLoadImages(images) {
      for(l=0; images.length > l; l++){
          var image = images[l];
          if (typeof image.dataset.lazySrc != "undefined") {
            if (image.src != image.dataset.lazySrc) {
              image.src = image.dataset.lazySrc;
            }
          }
        }
